Crea la configuraciĂłn de Nginx que termina TLS, sirve el frontend en `/` y proxea `/api/` al backend.

1. `infra/nginx/nginx.conf`:
```nginx
worker_processes auto;
events { worker_connections 1024; }

http {
  include /etc/nginx/mime.types;
  default_type application/octet-stream;
  sendfile on;
  keepalive_timeout 65;

  gzip on;
  gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

  upstream frontend_upstream { server frontend:3000; }
  upstream backend_upstream { server backend:3001; }

  server {
    listen 80;
    server_name _;
    return 301 https://$host$request_uri;
  }

  server {
    listen 443 ssl;
    server_name _;

    ssl_certificate     /etc/nginx/certs/lexscribe.crt;
    ssl_certificate_key /etc/nginx/certs/lexscribe.key;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         HIGH:!aNULL:!MD5;

    client_max_body_size 50m;

    location /api/ {
      proxy_pass         http://backend_upstream;
      proxy_http_version 1.1;
      proxy_set_header   Host $host;
      proxy_set_header   X-Real-IP $remote_addr;
      proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header   X-Forwarded-Proto $scheme;
    }

    location / {
      proxy_pass         http://frontend_upstream;
      proxy_http_version 1.1;
      proxy_set_header   Host $host;
      proxy_set_header   Upgrade $http_upgrade;
      proxy_set_header   Connection "upgrade";
      proxy_set_header   X-Real-IP $remote_addr;
      proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header   X-Forwarded-Proto $scheme;
    }
  }
}
```

4. `infra/scripts/generate-self-signed-cert.sh`:
```bash
#!/usr/bin/env bash
set -euo pipefail
CERT_DIR="$(dirname "$0")/../nginx/certs"
mkdir -p "$CERT_DIR"
openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout "$CERT_DIR/lexscribe.key" \
  -out    "$CERT_DIR/lexscribe.crt" \
  -subj "/CN=lexscribe.local"
echo "Self-signed cert generated at $CERT_DIR"
```
Make it executable: ensure file ends with newline; the script will be chmod +x by the developer locally.
