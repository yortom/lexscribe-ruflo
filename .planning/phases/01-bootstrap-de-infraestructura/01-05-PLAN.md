# Plan 01-05: GitHub Actions — pr.yml + deploy-staging.yml + deploy-prod.yml

## Objective
Tres pipelines de CI/CD: (1) `pr.yml` ejecuta lint + type-check + tests + build dry-run en cada PR a `main`; (2) `deploy-staging.yml` se dispara en merge a `main`, construye y publica imágenes Docker en GHCR y notifica al NAS para hacer pull; (3) `deploy-prod.yml` se dispara en tag `v*` y hace lo mismo apuntando a producción.

```yaml
name: PR Checks

on:
  pull_request:
    branches: [main]

concurrency:
  group: pr-${{ github.ref }}
  cancel-in-progress: true

jobs:
  checks:
    name: Lint + TypeCheck + Test + Build
    runs-on: ubuntu-latest
    timeout-minutes: 20
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 9.12.0

      - name: Setup Node 22
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm -r run lint

      - name: Type check
        run: pnpm -r run type-check

      - name: Test
        run: pnpm -r run test

      - name: Build (dry-run)
        run: pnpm -r --filter "./apps/*" run build
```
</action>

<action>
Crea `.github/workflows/deploy-staging.yml`. Construye imágenes para backend, frontend y nginx, las publica en GHCR taggeadas con `staging` y el SHA corto, y dispara un webhook al NAS para que ejecute `docker compose pull && up -d`.

Required GitHub repository secrets (to be set by user, NOT in repo):
- `NAS_STAGING_WEBHOOK_URL` — URL HTTPS al endpoint del NAS que ejecuta el comando deploy
- `NAS_STAGING_WEBHOOK_TOKEN` — bearer token para autenticar el webhook

`.github/workflows/deploy-staging.yml`:
```yaml
name: Deploy Staging

on:
  push:
    branches: [main]

concurrency:
  group: deploy-staging
  cancel-in-progress: false

env:
  REGISTRY: ghcr.io
  IMAGE_OWNER: ${{ github.repository_owner }}
  IMAGE_PREFIX: lexscribe

jobs:
  ci:
    name: Lint + TypeCheck + Test + Build
    runs-on: ubuntu-latest
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with: { version: 9.12.0 }
      - uses: actions/setup-node@v4
        with: { node-version: 22, cache: pnpm }
      - run: pnpm install --frozen-lockfile
      - run: pnpm -r run lint
      - run: pnpm -r run type-check
      - run: pnpm -r run test
      - run: pnpm -r --filter "./apps/*" run build

  build-and-push:
    name: Build & Push images to GHCR
    needs: ci
    runs-on: ubuntu-latest
    timeout-minutes: 30
    permissions:
      contents: read
      packages: write
    outputs:
      sha_tag: ${{ steps.meta.outputs.sha_tag }}
    steps:
      - uses: actions/checkout@v4

      - name: Compute short SHA
        id: meta
        run: echo "sha_tag=$(git rev-parse --short HEAD)" >> "$GITHUB_OUTPUT"

      - uses: docker/setup-buildx-action@v3

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build & push backend
        uses: docker/build-push-action@v6
        with:
          context: .
          file: apps/backend/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-backend:staging
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-backend:${{ steps.meta.outputs.sha_tag }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Build & push frontend
        uses: docker/build-push-action@v6
        with:
          context: .
          file: apps/frontend/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-frontend:staging
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-frontend:${{ steps.meta.outputs.sha_tag }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Build & push nginx
        uses: docker/build-push-action@v6
        with:
          context: infra/nginx
          file: infra/nginx/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-nginx:staging
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-nginx:${{ steps.meta.outputs.sha_tag }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  notify-nas:
    name: Notify NAS staging
    needs: build-and-push
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - name: Trigger NAS staging webhook
        run: |
          curl --fail --show-error --silent \
            -X POST "${{ secrets.NAS_STAGING_WEBHOOK_URL }}" \
            -H "Authorization: Bearer ${{ secrets.NAS_STAGING_WEBHOOK_TOKEN }}" \
            -H "Content-Type: application/json" \
            -d '{"environment":"staging","sha":"${{ needs.build-and-push.outputs.sha_tag }}"}'
```
</action>

`.github/workflows/deploy-prod.yml`:
```yaml
name: Deploy Production

on:
  push:
    tags:
      - 'v*'

concurrency:
  group: deploy-prod
  cancel-in-progress: false

env:
  REGISTRY: ghcr.io
  IMAGE_OWNER: ${{ github.repository_owner }}
  IMAGE_PREFIX: lexscribe

jobs:
  ci:
    name: Lint + TypeCheck + Test + Build
    runs-on: ubuntu-latest
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with: { version: 9.12.0 }
      - uses: actions/setup-node@v4
        with: { node-version: 22, cache: pnpm }
      - run: pnpm install --frozen-lockfile
      - run: pnpm -r run lint
      - run: pnpm -r run type-check
      - run: pnpm -r run test
      - run: pnpm -r --filter "./apps/*" run build

  build-and-push:
    name: Build & Push images to GHCR (prod)
    needs: ci
    runs-on: ubuntu-latest
    timeout-minutes: 30
    permissions:
      contents: read
      packages: write
    outputs:
      version: ${{ steps.meta.outputs.version }}
    steps:
      - uses: actions/checkout@v4

      - name: Extract version from tag
        id: meta
        run: echo "version=${GITHUB_REF_NAME}" >> "$GITHUB_OUTPUT"

      - uses: docker/setup-buildx-action@v3

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build & push backend
        uses: docker/build-push-action@v6
        with:
          context: .
          file: apps/backend/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-backend:prod
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-backend:${{ steps.meta.outputs.version }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Build & push frontend
        uses: docker/build-push-action@v6
        with:
          context: .
          file: apps/frontend/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-frontend:prod
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-frontend:${{ steps.meta.outputs.version }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Build & push nginx
        uses: docker/build-push-action@v6
        with:
          context: infra/nginx
          file: infra/nginx/Dockerfile
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-nginx:prod
            ${{ env.REGISTRY }}/${{ env.IMAGE_OWNER }}/${{ env.IMAGE_PREFIX }}-nginx:${{ steps.meta.outputs.version }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  notify-nas:
    name: Notify NAS production
    needs: build-and-push
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - name: Trigger NAS prod webhook
        run: |
          curl --fail --show-error --silent \
            -X POST "${{ secrets.NAS_PROD_WEBHOOK_URL }}" \
            -H "Authorization: Bearer ${{ secrets.NAS_PROD_WEBHOOK_TOKEN }}" \
            -H "Content-Type: application/json" \
            -d '{"environment":"production","version":"${{ needs.build-and-push.outputs.version }}"}'

  notify-deployment:
    name: Send deployment notification
    needs: [build-and-push, notify-nas]
    runs-on: ubuntu-latest
    timeout-minutes: 5
    if: always()
    steps:
      - name: Post deployment status
        if: ${{ secrets.DEPLOY_NOTIFICATION_WEBHOOK != '' }}
        run: |
          STATUS="${{ needs.notify-nas.result }}"
          curl --fail --show-error --silent \
            -X POST "${{ secrets.DEPLOY_NOTIFICATION_WEBHOOK }}" \
            -H "Content-Type: application/json" \
            -d "{\"text\":\"Lexscribe prod deploy ${{ needs.build-and-push.outputs.version }}: ${STATUS}\"}"
```
